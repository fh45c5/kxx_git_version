RRM_increase_chaos = {
  if = {
    limit = {
      has_idea = RRM_chaos_low
    }
    swap_ideas = {
      remove_idea = RRM_chaos_low
      add_idea = RRM_chaos_medium
    }
    else = {
      if = {
        limit = {
          has_idea = RRM_chaos_medium
        }
        swap_ideas = {
          remove_idea = RRM_chaos_medium
          add_idea = RRM_chaos_high
        }
        else = {
          if = {
            limit = {
              has_idea = RRM_chaos_high
            }
            swap_ideas = {
              remove_idea = RRM_chaos_high
              add_idea = RRM_anarchy
            }
          }
        }
      }
    }
  }
}
RRM_decrease_chaos = {

}
RRM_check_if_chaos_idea_needs_adjustment = {
  if = {
    limit = {
      has_idea = RRM_chaos_low
      check_variable = {
        chaos > 3
      }
    }
    hidden_effect = {
      country_event = {
        id = royalist.9
        days = 1
      }
    }
  }
  if = {
    limit = {
      has_idea = RRM_chaos_medium
      check_variable = {
        chaos > 6
      }
    }
    hidden_effect = {
      country_event = {
        id = royalist.9
        days = 1
      }
    }
  }
  if = {
    limit = {
      has_idea = RRM_chaos_medium
      check_variable = {
        chaos < 4
      }
    }
    hidden_effect = {
      country_event = {
        id = royalist.10
        days = 1
      }
    }
  }
  if = {
    limit = {
      has_idea = RRM_chaos_high
      check_variable = {
        chaos > 9
      }
    }
    hidden_effect = {
      country_event = {
        id = royalist.11
        days = 1
      }
    }
  }
  if = {
    limit = {
      has_idea = RRM_chaos_high
      check_variable = {
        chaos < 7
      }
    }
    hidden_effect = {
      country_event = {
        id = royalist.10
        days = 1
      }
    }
  }
  if = {
    limit = {
      has_idea = RRM_anarchy
      check_variable = {
        chaos < 10
      }
    }
    hidden_effect = {
      country_event = {
        id = royalist.10
        days = 1
      }
    }
  }
}
RRM_increase_chaos_by_1 = {
  add_to_variable = {
    var = chaos
    value = 1
  }
  RRM_check_if_chaos_idea_needs_adjustment = yes
  custom_effect_tooltip = RRM_increases_chaos_slightly
